\chapter{Giới Thiệu Đề Tài}

\section{Giới thiệu}
Đề tài xây dựng hệ thống thương mại điện tử bán voucher giảm giá trực tuyến. Hệ thống cần hỗ trợ ba vai trò: khách hàng, đối tác và quản trị viên. Tài liệu BRD yêu cầu các luồng mua voucher, phát hành code, xác nhận sử dụng, báo cáo, quản lý nội dung và kiểm toán.

\section{Bối Cảnh Và Mục Tiêu}
Mục tiêu hệ thống là cung cấp kênh bán voucher trực tuyến có quy trình duyệt rõ ràng, kiểm soát tồn kho và trạng thái, có phân quyền và có khả năng báo cáo quản trị.

\section{Phạm Vi Hệ Thống}
\begin{itemize}[leftmargin=1.2cm]
  \item Khách hàng: tìm kiếm, giỏ hàng, mua, thanh toán, nhận voucher code, đánh giá và khiếu nại.
  \item Đối tác: đăng ký hồ sơ, tạo voucher, gửi duyệt, quản lý voucher, kiểm tra và xác nhận sử dụng.
  \item Quản trị viên: duyệt đối tác, duyệt voucher, quản lý đơn hàng, nội dung, dashboard và nhật ký.
\end{itemize}

\subsection{Công nghệ Chi tiết}

\begin{itemize}[leftmargin=1.2cm]
  \item \textbf{Backend}: Node.js, Express.js, PostgreSQL (pg driver), JWT (jsonwebtoken), bcryptjs, UUID, Morgan (logging), CORS.
  \item \textbf{Frontend}: React 18, React Router DOM v6, Axios, react-icons, react-slider, CSS-in-JS.
  \item \textbf{Authentication}: JWT bearer tokens, bcryptjs password hashing, role-based authorization (ADMIN, PARTNER, CUSTOMER).
  \item \textbf{Payment Integration}: VNPay (IPN callback), PayPal SDK (Sandbox), thanh toán mô phỏng (COD).
  \item \textbf{Database}: PostgreSQL schema với enums (user\_role, voucher\_status, order\_status, issued\_status), constraints, indexes, foreign keys.
\end{itemize}

\section{Các Lỗi Thường Gặp Và Cách Khắc Phục}

\subsection{Docker Không Khởi Động Được}

\textbf{Triệu chứng:} \texttt{docker compose up} báo lỗi, services không chạy.

\textbf{Nguyên nhân phổ biến:}
\begin{itemize}[leftmargin=1.2cm]
  \item Docker Desktop chưa được khởi động.
  \item Cổng 3000, 5001, 5499 đã bị chiếm dụng.
  \item File \texttt{.env} chưa được tạo.
\end{itemize}

\textbf{Giải pháp:}
\begin{itemize}[leftmargin=1.2cm]
  \item Kiểm tra Docker Desktop đang chạy: \texttt{docker ps}.
  \item Xem quy trình sử dụng cổng: \codepath{lsof -i :3000} (macOS/Linux) hoặc \codepath{netstat -ano | findstr :3000} (Windows).
  \item Tạo file \texttt{.env} từ \texttt{.env.example}: \texttt{cp .env.example .env}.
  \item Reset Docker: \texttt{docker compose down -v; docker compose up --build}.
\end{itemize}

\subsection{Frontend Không Kết Nối Được API}

\textbf{Triệu chứng:} Frontend hiển thị lỗi kết nối, API call không thành công.

\textbf{Nguyên nhân:}
\begin{itemize}[leftmargin=1.2cm]
  \item Backend chưa khởi động.
  \item CORS không được cấu hình.
  \item Cổng API sai trong frontend config.
\end{itemize}

\textbf{Giải pháp:}
\begin{itemize}[leftmargin=1.5cm]
  \item Kiểm tra backend chạy: \codepath{curl http://localhost:5001/health}.
  \item Kiểm tra CORS trong \codepath{backend/src/app.js} có chứa \codepath{http://localhost:3000}.
  \item Kiểm tra file \codepath{frontend/src/services/api.js} có baseURL đúng.
\end{itemize}

\subsection{Database Không Kết Nối}

\textbf{Triệu chứng:} Backend báo lỗi \codepath{ECONNREFUSED} hoặc \texttt{database connection failed}.

\textbf{Nguyên nhân:}
\begin{itemize}[leftmargin=1.2cm]
  \item PostgreSQL container chưa sẵn sàng.
  \item \codepath{DATABASE_URL} sai trong \texttt{.env}.
  \item Dữ liệu PostgreSQL bị hỏng.
\end{itemize}

\textbf{Giải pháp:}
\begin{itemize}[leftmargin=1.5cm]
  \item Kiểm tra container: \texttt{docker compose ps}.
  \item Kiểm tra logs: \texttt{docker compose logs db}.
  \item Reset: \texttt{docker compose down -v; docker compose up --build}.
  \item Nếu chạy native: \texttt{psql -U voucheruser -d voucherdb -c "\textbackslash dt"} để liệt kê bảng.
\end{itemize}

\subsection{Token JWT Hết Hạn (401 Unauthorized)}

\textbf{Triệu chứng:} Sau vài giờ, bị logout tự động.

\textbf{Nguyên nhân:}
\begin{itemize}[leftmargin=1.2cm]
  \item JWT token hết hạn (mặc định 7 ngày theo \codepath{JWT_EXPIRES_IN} trong \texttt{.env}).
\end{itemize}

\textbf{Giải pháp:}
\begin{itemize}[leftmargin=1.5cm]
  \item Đăng nhập lại.
  \item Nếu cần token sống lâu hơn, chỉnh \codepath{JWT_EXPIRES_IN} trong \texttt{.env} (ví dụ: \texttt{30d}).
\end{itemize}

\subsection{Package Dependencies Lỗi}

\textbf{Triệu chứng:} \texttt{npm install} báo conflict, hoặc app khởi động với cảnh báo.

\textbf{Giải pháp:}
\begin{itemize}[leftmargin=1.5cm]
  \item Xóa \codepath{package-lock.json} và \codepath{node_modules}: \codepath{rm -rf node_modules package-lock.json} (Linux/macOS) hoặc \codepath{rmdir /s node_modules & del package-lock.json} (Windows).
  \item Cài lại: \texttt{npm install}.
  \item Nếu vẫn lỗi, dùng \texttt{npm ci} thay vì \texttt{npm install}.
\end{itemize}
